import React, { useRef, useState } from 'react';
import axios from 'axios';
import './App.css';

function App() {
  const [file, setFile] = useState(null);
  const audioContext = new (window.AudioContext || window.webkitAudioContext)();
  const sourceNodeRef = useRef(null);
  const filterNodeRef = useRef(audioContext.createBiquadFilter());
  const gainNodeRef = useRef(audioContext.createGain());

  useEffect(() => {
    filterNodeRef.current.type = 'lowpass';
    filterNodeRef.current.frequency.setValueAtTime(1000, audioContext.currentTime);

    gainNodeRef.current.gain.setValueAtTime(1, audioContext.currentTime);

    filterNodeRef.current.connect(gainNodeRef.current);
    gainNodeRef.current.connect(audioContext.destination);
  }, []);

  const handleFileChange = (event) => {
    setFile(event.target.files[0]);
  };

  const handlePlay = async () => {
    if (!file) return;

    const formData = new FormData();
    formData.append('file', file);

    try {
      const response = await axios.post('/upload', formData, {
        headers: {
          'Content-Type': 'multipart/form-data'
        }
      });

      const audioBuffer = await fetch(response.data.url)
        .then(res => res.arrayBuffer())
        .then(arrayBuffer => audioContext.decodeAudioData(arrayBuffer));

      sourceNodeRef.current?.disconnect();
      sourceNodeRef.current = audioContext.createBufferSource();
      sourceNodeRef.current.buffer = audioBuffer;
      sourceNodeRef.current.connect(filterNodeRef.current);
      sourceNodeRef.current.playbackRate.setValueAtTime(1, audioContext.currentTime);
      sourceNodeRef.current.start();
    } catch (error) {
      console.error('Error uploading file:', error);
    }
  };

  const handleStop = () => {
    sourceNodeRef.current?.stop();
  };

  const handleVolumeChange = (event) => {
    gainNodeRef.current.gain.setValueAtTime(event.target.value, audioContext.currentTime);
  };

  const handlePlaybackRateChange = (event) => {
    sourceNodeRef.current.playbackRate.setValueAtTime(event.target.value, audioContext.currentTime);
  };

  const handleFilterFrequencyChange = (event) => {
    filterNodeRef.current.frequency.setValueAtTime(event.target.value, audioContext.currentTime);
  };

  return (
    <div className="App bg-gray-900 text-white min-h-screen flex items-center justify-center">
      <div className="p-8 rounded-lg shadow-lg bg-gray-800 w-full max-w-3xl">
        <h1 className="text-2xl font-bold mb-4">Audio Workstation</h1>
        <input type="file" accept=".mp3" onChange={handleFileChange} className="mb-4"/>
        <button onClick={handlePlay} className="bg-blue-500 text-white px-4 py-2 rounded mr-2">Play</button>
        <button onClick={handleStop} className="bg-red-500 text-white px-4 py-2 rounded">Stop</button>
        <div className="mt-4">
          <label>Volume:</label>
          <input type="range" min="0" max="1" step="0.01" defaultValue="1" onChange={handleVolumeChange} className="w-full"/>
        </div>
        <div className="mt-4">
          <label>Playback Rate:</label>
          <input type="range" min="0.5" max="2" step="0.01" defaultValue="1" onChange={handlePlaybackRateChange} className="w-full"/>
        </div>
        <div className="mt-4">
          <label>Filter Frequency (Hz):</label>
          <input type="range" min="100" max="20000" step="100" defaultValue="1000" onChange={handleFilterFrequencyChange} className="w-full"/>
        </div>
      </div>
    </div>
  );
}

export default App;

from fastapi import FastAPI, File, UploadFile
from fastapi.responses import JSONResponse
import os
from uuid import uuid4

app = FastAPI()

@app.get("/")
async def read_root():
    return {"message": "Welcome to the Audio Workstation API"}

@app.get("/health")
async def health_check():
    return {"status": "ok"}

@app.post("/upload/")
async def upload_file(file: UploadFile = File(...)):
    file_location = f"uploads/{uuid4()}-{file.filename}"
    with open(file_location, "wb+") as file_object:
        file_object.write(file.file.read())
    return JSONResponse(content={"url": file_location})

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)

#!/bin/bash

case "$1" in
  start)
    ./start.sh
    ;;
  stop)
    ./stop.sh
    ;;
  restart)
    ./restart.sh
    ;;
  *)
    echo "Usage: $0 {start|stop|restart}"
    exit 1
    ;;
esac

exit 0
